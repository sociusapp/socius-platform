# Required packages for React Native App Update System

# For device info and version checking
npm install react-native-device-info

# For downloading APK files
npm install rn-fetch-blob

# For file operations (optional)
npm install react-native-fs

# For network state checking (optional)
npm install @react-native-community/netinfo

# For permissions (Android)
npm install react-native-permissions
